Input Format:
    First line contain 2 integer - number of node and number of edges
    then next lines contain u and v and u is connected to v

Representation:
    Adjacency Matrix
    Adjacency List

Adjacency Matrix:
    A 2D array of size n*n where n is number of nodes a[i][j] = 1 if i is connected to j

    c++ code:
        int main()
        {
            int n, m;
            cin>>n>>m;
            int a[n+1][m+1];
            for(int i = 0; i < m; i++)
            {
                int u, v;
                cin>>u>>v;
                a[u][v] = 1;
                a[v][u] = 1; // It will be removed in case of directed graph
            }
            return 0;
        }

Adjacency List:
    Optimised as previous one takes O(n^2) space
    It is node based Representation.
    Each node with a list of nodes to which it is connected

    Here space : O(2*E) where E is number of edges

    c++ code:
        int main()
        {
            int n, m;
            cin>>n>>m;
            vector<vector<int>> a(n+1);
            for(int i = 0; i < m; i++)
            {
                int u, v;
                cin>>u>>v;
                a[u].push_back(v);
                a[v].push_back(u);
            }
            return 0;
        }

Weighted Graph Representation:
    vector<vector< pair<int, int> >> a(n + 1);

    Undirected Graph
    int u, v, wt;
    cin >> u >> v >> wt;
    adj[u][v] = wt;
    adj[v][u] = wt;

    Directed Graph
    int u, v, wt;
    cin >> u >> v >> wt;
    adj[u][v] = wt;

    c++ code:
        int main()
        {
            int n, m;
            cin >> n >> m;

            vector<vector<pair<int,int>>> a(n + 1);

            for(int i = 0; i < m; i++)
            {
                int u, v, w;
                cin >> u >> v >> w;

                a[u].push_back({v, w});
                a[v].push_back({u, w}); // remove if graph is directed
            }

            return 0;
        }
